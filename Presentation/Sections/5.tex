\section{Merkle Trees}

\begin{frame}{Na\"\i ve Commitment}
    \begin{itemize}
        \item Let the hidden board state be
        \begin{equation*}
            x = \begin{pvector}
                x_1 & \dots & x_N
            \end{pvector},
        \end{equation*}
        where each $x_j \in \bin$ indicates whether square $j$ contains a ship.
    
        \item At setup, the defender publishes a commitment to the whole board:
        \begin{equation*}
            C = \hash\del{x_1, \dots, x_N}.
        \end{equation*}
    
        \item For a shot at square $i$, the defender reports
        \begin{equation*}
            y \in \set{\hit, \miss}.
        \end{equation*}
    
        \item The defender proves
        \begin{align*}
            &\exists x_1, \dots, x_N \\
            \text{s.t.} \quad
            & C = \hash\del{x_1, \dots, x_N}, \\
            & y = f\del{x_i}.
        \end{align*}
    \end{itemize}
\end{frame}

\begin{frame}{Na\"\i ve Commitment: Cost}
    \begin{itemize}
        \item \defn{Problem:} Even though we only need to verify the local value $x_i$, the circuit must re-hash the entire board.
    
        \item $\pcalgostyle{Groth16}$ proving cost depends on the number of circuit constraints.
    
        \item Re-hashing the full board creates constraints that scale with the size of the game state:
        \begin{equation*}
            \bigO{N}.
        \end{equation*}
    \end{itemize}
\end{frame}

\begin{frame}{Merkle Tree Commitment}
    \small
    \begin{itemize}
        \item \defn{Goal:} Commit to many values using one root hash.
    
        \item In our case, the values are the board squares $x_1, \dots, x_N$.
    
        \item The defender publishes the Merkle root
        \begin{equation*}
            R = \pcalgostyle{MerkleRoot}\del{\ell_1, \dots, \ell_N},
        \end{equation*}
        
        where each leaf
        \begin{equation*}
            \ell_j = \hash_{\pcalgostyle{leaf}}\del{j, x_j, r_j}
        \end{equation*}
        
        is a commitment to square $j$.

        \item Here:
        \begin{align*}
            x_j &= \text{value of square } j, \\
            r_j &= \text{private randomness}.
        \end{align*}
    
        \item $R$ commits to the entire board, but a square can be \\
        verified using only its \defn{Merkle path}.
    \end{itemize}
\end{frame}

            \toprule
                                   & \thead{Na\"\i ve}  & \thead{Merkle}          \\
            \midrule
            \thead{Private inputs} & $x_1, \dots, x_N$  & $x_i, \; r_i, \; \Pi_i$ \\
            \thead{Commitment}     & $C$                & $R$                     \\
            \thead{Constraints}    & $\bigO{N}$         & $\bigO{\log N}$         \\
            \bottomrule
        \end{tblr}
